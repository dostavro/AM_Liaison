def unsubscribe(host)
  pubsub.subscriptions(prefix_host(host)) do |m|
    m[:subscribed] && m[:subscribed].each do |s|
      pubsub.unsubscribe(s[:node], nil, s[:subid], prefix_host(host), &callback_logging(__method__, s[:node], s[:subid]))
    end
  end
end
