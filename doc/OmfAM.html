def initialize(urn, uri)
  super(urn, URI.parse(uri))
  
  @comm = OmfCommon::Comm.new('xmpp')

  @comm.when_ready do
    logger.info "CONNECTED: #{@comm.jid.inspect}"
    
    @comm.subscribe(urn, @uri.host) 
  end
  @comm.connect('broker', 'pw', @uri.host)
end
